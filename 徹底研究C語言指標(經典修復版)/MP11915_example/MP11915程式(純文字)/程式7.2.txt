/*
** 癸榷i姘府张纭C
*/

int
even_parity( int value, int n_bits )
{
        int parity = 0;

        /*
        ** p衡痊1害欷腑蛹啤C
        */
        while( n_bits > 0 ){
                parity += value & 1;
                value >>= 1;
                n_bits -= 1;
        }

        /*
        ** pGp计竟撼抬C欷脯O0A
        ** 穰^TRUE(恁1害欷讣片鞍讣骗)C
        */
        return ( parity % 2 ) == 0;
}
