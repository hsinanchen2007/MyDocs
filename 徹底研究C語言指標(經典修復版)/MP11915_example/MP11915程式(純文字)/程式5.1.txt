int
count_one_bits( unsigned value )
{
        int ones;

        /*
        ** 讽临Τ@ㄇ欷釜涵痊1...
        */
        for( ones = 0; value != 0; value = value >> 1 )
                /*
                ** pG程C欷釜涵痊1Ap计W[1C
                */
                if( value % 2 != 0 )
                        ones = ones + 1;

        return ones;
}
