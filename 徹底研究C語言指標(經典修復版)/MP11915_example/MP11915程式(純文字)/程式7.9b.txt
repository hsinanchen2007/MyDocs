#include <stdarg.h>

float
average( int n_values, ... )
{
        va_list   var_arg;
        int    count;
        float sum = 0;

        /*
        ** 非称si跑把计C
        */
        va_start( var_arg, n_values );

        /*
        ** K[邾i跑把计C愍涵取C
        */
        for( count = 0; count < n_values; count += 1 ){
                sum += va_arg( var_arg, int );
        }

        /*
        ** ЧΘBzi跑把计C
        */
        va_end( var_arg );

        return sum / n_values;
}
