/*
** 從一個字元陣列中提取空白字元分隔 
** 的token並把它們列印出來(每列一個)。
*/
#include <stdio.h>
#include <string.h>

void
print_tokens( char *line )
{
        static char whitespace[] = " \t\f\r\v\n";
        char   *token;

        for( token = strtok( line, whitespace );
            token != NULL;
            token = strtok( NULL, whitespace ) )
                printf( "Next token is %s\n", token );
}
