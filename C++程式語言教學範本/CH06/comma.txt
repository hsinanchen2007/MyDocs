//comma.cpp
#include <iostream>
using namespace std;

int main()
{
    int x, y, z;

    cout << "  x   y" << endl;
    for(x=1, y=10; x<=6 && y<=60; x++, y=y+10)
        cout << "  " << x << "  " << y << endl;
    cout << endl;

    z = (x=100, (y=1000)+14);
    cout << "x = " << x << " y = " << y << " z = " << z << endl;

    return 0;
}
