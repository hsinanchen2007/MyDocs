//while99.cpp
#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int i,j;

    i = 1;
        
    while(i <= 9) {
        j = 1;
        while (j <= 9) {
            cout << setw(4) << i*j;
            j++;
        }
        cout << endl;
        i++;
    }

    return 0;
}
