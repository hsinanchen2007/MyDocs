//passwd.cpp
#include <iostream>
#define PASSWD 1462
#define TRUE 1
#define FALSE 0
using namespace std;

int main()
{
    int passwd;
    int ok, tryTimes;
    ok = FALSE;
    tryTimes = 0;

    do {
        cout << ++tryTimes << ". Enter your password: ";
        cin >> passwd;
        if (passwd == PASSWD)
            ok = TRUE;
    } while (!ok && (tryTimes < 3));

    if (ok)
        cout << endl << "Congratulations!" << endl;
    else
        cout << endl << "You are rejected!" << endl;

    return 0;
}
