//divide.cpp
#include<iostream>
#include <iomanip>
using namespace std;

int main()
{
    int op1 = 10;
    int op2 = 4;
    double op3 = 10.0;
    double op4 = 4.0;

    cout << fixed << setprecision(2);
    cout << "Divide and Trunc ..." << endl << endl;
 
    cout << "    " << op1 << " / " << op2 << " = " << (op1/op2) << endl;
    cout << "    " << op1 << " / " << op4 << " = " << (op1/op4) << endl;
    cout << "    " << op3 << " / " << op4 << " = " << (op3/op4) << endl;
    cout << "    " << op3 << " / " << op2 << " = " << (op3/op2) << endl;

    return 0;
}
