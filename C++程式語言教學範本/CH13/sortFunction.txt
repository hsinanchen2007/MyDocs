//sortFunction.cpp
#include <iostream>
#define SIZE 10
using namespace std;
void bubbleSortInt(int [], int);
void bubbleSortDouble(double [], int);

double doubleData[] = {1.2, 3.2, 0.5, 2.2, 0.8};
int main()
{
    int num[SIZE];
    srand((unsigned int) time(NULL));
    cout << "Original data: " << endl;
    for (int i=0; i<SIZE; i++) {
        num[i] = rand() % 100 + 1;
        cout << num[i] << " ";
    }
    bubbleSortInt(num, SIZE);
    cout << "\nSorted data: " << endl;
    for (int j=0; j<SIZE; j++) {
        cout << num[j] << " ";
    }
    cout << endl << endl;
    
    //double data
    cout << "Original data: " << endl;
    for (int j=0; j<5; j++) {
        cout << doubleData[j] << " ";
    }
    bubbleSortDouble(doubleData,5);
    cout << "\nSorted data: " << endl;
    for (int j=0; j<5; j++) {
        cout << doubleData[j] << " ";
    }
    cout << endl;
    return 0;
}

void bubbleSortInt(int data[], int n)
{
    int temp;
    for (int i=0; i<n-1; i++) {
        for (int j=0; j<n-i-1; j++) {
            if (data[j] > data[j+1]) {
                temp = data[j];
                data[j] = data[j+1];
                data[j+1] = temp;
            }
        }
    }
}

void bubbleSortDouble(double data2[], int n)
{
    double temp;
    for (int i=0; i<n-1; i++) {
        for (int j=0; j<n-i-1; j++) {
            if (data2[j] > data2[j+1]) {
                temp = data2[j];
                data2[j] = data2[j+1];
                data2[j+1] = temp;
            }
        }
    }
}
