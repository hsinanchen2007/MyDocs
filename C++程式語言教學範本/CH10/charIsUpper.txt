//charIsUpper.cpp
#include <iostream>
using namespace std;

int main()
{
    string str = "BMW X6";
    int ans;
    int i;
    cout << "檢視字元是否為大寫英文字母" << endl;
    for (i=0; i<str.length(); i++) {
        ans = isupper(str[i]);
        cout << str[i] << ": " << ans << endl;
    }

    return 0;
}
