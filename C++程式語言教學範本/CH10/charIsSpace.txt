//charIsSpace.cpp
#include <iostream>
using namespace std;

int main()
{
    string str = "BMW X6";
    int ans;
    int i;
    cout << "檢視字元是否為空白" << endl;
    for (i=0; i<str.length(); i++) {
        ans = isspace(str[i]);
        cout << str[i] << ": " << ans << endl;
    }

    return 0;
}
