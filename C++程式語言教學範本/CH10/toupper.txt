//toupper.cpp
#include <iostream>
using namespace std;

int main()
{
    string str2 = "New BMW X6";
    char ans;
    int i;
    cout << "將字元轉換為大寫英文字母" << endl;
    for (i=0; i<str2.length(); i++) {
        ans = toupper(str2[i]);
        cout << str2[i] << ": " << ans << endl;
    }

    return 0;
}
