//conAndDest.cpp
#include <iostream>
#define SIZE 10
using namespace std;

class Array {
public:
    Array();
    void getData();
    int calTot();
    ~Array();

private:
    int *num;
};

Array::Array()
{
    cout << "執行建構函式..." << endl;
    num = new int[SIZE];
}

void Array::getData()
{
    for (int i=0; i<SIZE; i++) {
        num[i] = i+1;
        cout << "num" << "[" << i << "] = " << num[i]  << endl;
    }
}

int Array::calTot()
{
    int total=0;
    for (int i=0; i<SIZE; i++) {
        total += num[i];
    }
    return total;
}

Array::~Array()
{
   cout << "執行解構函式..." << endl;
    delete [] num;
}

int main()
{
    Array obj;
    obj.getData();
    cout << "總和: " << obj.calTot() << endl;
    return 0;
}
