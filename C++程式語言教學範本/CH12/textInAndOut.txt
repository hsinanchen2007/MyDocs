//textInAndOut.cpp
#include <iostream>
#include <fstream>
using namespace std;

int main()
{
    ofstream outFile;
    ifstream inFile;
    string id, name;
    double score;
    
    //write data into file
    outFile.open("score.dat");
    cout << "請輸入ID: ";
    cin >> id;
    cout << "請輸入姓名: ";
    cin >> name;
    cout << "請輸入分數: ";
    cin >> score;
    
    cout << "\nwriting ... " << endl;
    outFile << id << " " << name << " " << score << endl;
    outFile.close();
    cout << "close file" << endl << endl;
    
    //read data from file
    inFile.open("score.dat");
    cout << "reading ... " << endl;
    inFile >> id >> name >> score;
    cout << id << " " << name << " " << score << endl;
    
    inFile.close();
    cout << "close file" << endl;
    
    return 0;
}
