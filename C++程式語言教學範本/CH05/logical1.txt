//logical1.cpp
#include <iostream>
using namespace std;

int main()
{
    cout << "NOT (3 > 5) ===> " << !(3 > 5) << endl;
    cout << "(3 > 5) OR (10 > 6) ===> " << ((3 > 5) || (10 > 6)) << endl;
    cout << "(3 > 5) AND (10 > 6) ===> " << ((3 > 5) && (10 > 6)) << endl;
    cout << "(3 > 5) AND (10 > 6) OR 10 ===> " 
         << ((3 > 5) && (10 > 6) || 10) << endl;
    cout << "(5 > 3) OR (10 > 6) AND 0 ===> " 
         << ((5 > 3) || (10 > 6) && 0) << endl;

    return 0;
}
