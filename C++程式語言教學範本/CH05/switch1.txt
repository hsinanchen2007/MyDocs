//switch1.cpp
#include <iostream>
using namespace std;

int main()
{
    char grade;

    cout << "What's your score grade (A-E)? ";
    cin >> grade;

    if ((grade >= 'A') && (grade <= 'Z'))
        grade = 'a' + (grade - 'A');
    switch (grade) {
        case 'a': cout << "Score 90 to 100." << endl;
                  break;
        case 'b': cout << "Score 80 to 89." << endl;
                  break;
        case 'c': cout << "Score 70 to 79." << endl;
                  break;
        case 'd': cout << "Score 60 to 69." << endl;
                  break;
        case 'e': cout << "Score under 60." << endl;
                  break;
        default: cout << "Wrong grade !" << endl;
    }

    return 0;
}
