//elseIf.cpp
#include <iostream>
using namespace std;

int main()
{
    char grade;
    int score;

    cout << "What's your score? ";
    cin >> score;

    if(score > 100 || score < 0)
        cout << "It's impossible!" << endl;
    else if (score >= 90)
        grade = 'A';
    else if (score >= 80)
        grade = 'B';
    else if (score >= 70)
        grade = 'C';
    else if (score >= 60)
        grade = 'D';
    else
        cout << "You are down!" << endl;
    if( score >= 60 && score <=100)
        cout <<  "Score " << score << " ===> " << grade << endl;
    return 0;
}
