#include <iostream>
using namespace std;

int main()
{
    cout << "10 > 100 ===> " << (10 > 100) << endl;
    cout << "10 < 100 ===> " << (10 < 100) << endl;
    cout << "10 == 100 ===> " << (10 == 100) << endl;
    cout << "10 != 100 ===> " << (10 != 100) << endl;

    return 0;
}
