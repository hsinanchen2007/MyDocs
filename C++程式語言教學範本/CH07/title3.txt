#include <iostream>
void title();
void choose(char);
using namespace std;

int main()
{
    char ch;
    
    title();
    cout << "\n    Choose ===> ";
    cin >> ch;
    cout << "\n    Echo : ";
    choose(ch);
    cout << endl;

    title();
    cout << "\n    Choose ===> ";
    cin >> ch;
    cout << "\n    Echo : ";
    choose(ch);
    cout << endl;
    
    return 0;
}

void title()
{
    cout << "    ====MENU====\n";
    cout << " [1] one    [2] two\n";
    cout << " [3] three  [4] four\n";
}

void choose(char ch)
{
    switch (ch) {
        case '1':  cout << "ONE";
                    break;
        case '2':  cout << "TWO";
                    break;
        case '3':  cout << "THREE";
                    break;
        case '4':  cout << "FOUR";
    }
    cout << endl;
}
