//recursive.cpp
#include <iostream>
using namespace std;
void recur(int);

int main()
{
    cout << "Recursive call..." << endl << endl;
    recur(4);

    return 0;
}

void recur(int level)
{
    cout << "    BEFORE level ===> " << level << endl;
    if(level > 0)
        recur(level-1);
    cout << "    AFTER level ===> " << level << endl;
}
