//factorial.cpp
#include <iostream>
#include <iomanip>
using namespace std;
unsigned int fact(int);

int main()
{
    int i;

    cout << "Factorial..." << endl << endl;
    for(i=0; i<=17; i++)
        cout << "    " << setw(3) << i << "! = "
             << fact(i) << endl;

    return 0;
}

unsigned int fact(int num)
{
    if (num == 0)
        return(1);
    return (num * fact (num-1));
}
