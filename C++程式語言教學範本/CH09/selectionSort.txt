#include <iostream>
#include <iomanip>
using namespace std;

#define N 10
void list(int *);
void swap(int *, int *);
void selectionSort(int *);

int main()
{
    int i,arr[N];
    srand(unsigned(time(NULL)));
    for (i=0; i<N; i++)
        arr[i] = rand() % 100 + 1;
          
    cout << "\nSource array ...\n";
    list(arr);
       
    selectionSort(arr);
    cout << "Sorting ...\n";
    list(arr);

    return 0;
}

void list(int *array)
{
    int i;
   
    for (i=0; i<N; i++) {
        cout << setw(5) << array[i];
        if (((i+1) % 5) == 0)
            cout << endl;
    }
    cout << endl;
}
     
void swap(int *i, int *j)
{
    int temp;
    
    temp = *i;
    *i = *j;
    *j = temp;
}
    
void selectionSort(int *arr2)
{
    int i, j, min;
     
    for (i=0; i<N; i++) {
        min = i;
        for (j=i; j<N; j++)
            if (arr2[j] < arr2[min])
                min = j;
        swap(&arr2[min], &arr2[i]);
    }
}
