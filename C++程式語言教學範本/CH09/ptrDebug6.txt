#include <iostream>
using namespace std;

int sum(int *[], int, int);
int main()
{
    int arr2[2][3] = {{10, 20, 30}, {40, 50, 60}};
    cout << "將二維陣列 arr 加總後為: " << sum(arr2, 2, 3) << endl;
    
    return 0;
}

int sum(int *arr2[], int row, int col)
{
    int i, j, total = 0;
    for (i=0; i<row; i++)
        for (j=0; j<col; j++)
            total += arr2[i][j];
    return total;
}
