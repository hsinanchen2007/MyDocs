//friendFunction.cpp
#include <iostream>
using namespace std;

class Appointment
{
public:
    Appointment(int, int, int);    
    friend void display();

private:
    int year;
    int month;
    int day;
};

Appointment::Appointment(int year, int month, int day)
{
    this->year = year;
    this->month = month;
    this->day = day;
}

void display()
{
    Appointment obj(2020, 6, 6);
    cout << obj.year << "/" << obj.month << "/" << obj.day << endl;
}

int main()
{
    display();
    return 0;
}
